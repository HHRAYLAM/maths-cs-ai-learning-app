**规划算法**：

```python
# Tom 的任务规划器
class TaskPlanner:
    def __init__(self, llm):
        self.llm = llm  # 大语言模型
    
    def decompose(self, goal, max_depth=3):
        """
        将目标分解为子任务
        
        Tom 的递归分解法：
        大任务 → 中任务 → 小任务 → 可执行动作
        """
        if max_depth == 0:
            return [goal]
        
        # 用 LLM 分解任务
        prompt = f"""
        将以下任务分解为可执行的子任务：
        任务：{goal}
        
        返回子任务列表，每个子任务应该是：
        1. 具体的
        2. 可执行的
        3. 有明确完成标准的
        """
        
        subtasks = self.llm.generate(prompt)
        
        # 递归分解每个子任务
        result = []
        for subtask in subtasks:
            if self.is_complex(subtask):
                result.extend(self.decompose(subtask, max_depth - 1))
            else:
                result.append(subtask)
        
        return result
    
    def is_complex(self, task):
        """判断任务是否复杂，需要进一步分解"""
        # 简化判断：看任务描述长度
        return len(task) > 50
```

```python
import json

# Tom 的工具注册表
class ToolRegistry:
    def __init__(self):
        self.tools = {}
    
    def register(self, name, description, func):
        """注册一个工具"""
        self.tools[name] = {
            'description': description,
            'function': func
        }
    
    def get_tool_schema(self, name):
        """获取工具描述（用于 LLM 理解）"""
        return {
            'name': name,
            'description': self.tools[name]['description']
        }
    
    def call(self, name, **kwargs):
        """调用工具"""
        if name not in self.tools:
            raise ValueError(f"未知工具：{name}")
        return self.tools[name]['function'](**kwargs)

# 注册工具
registry = ToolRegistry()

registry.register(
    'search',
    '在互联网上搜索信息',
    lambda query: f"搜索结果：关于'{query}'的信息..."
)

registry.register(
    'calculate',
    '执行数学计算',
    lambda expression: f"计算结果：{eval(expression)}"
)

registry.register(
    'get_weather',
    '查询天气',
    lambda location: f"{location}的天气：晴朗，25°C"
)

# Tom 的智能体工具调用
class AgentWithTools:
    def __init__(self, llm, tool_registry):
        self.llm = llm
        self.tools = tool_registry
    
    def execute(self, goal):
        """执行目标"""
        # 1. 让 LLM 决定用什么工具
        prompt = f"""
        目标：{goal}
        
        可用工具：
        {json.dumps([self.tools.get_tool_schema(name) 
                     for name in self.tools.tools.keys()], indent=2)}
        
        请选择需要调用的工具和参数。
        """
        
        tool_call = self.llm.generate(prompt)
        
        # 2. 解析工具调用
        tool_name, tool_args = self.parse_tool_call(tool_call)
        
        # 3. 执行工具
        result = self.tools.call(tool_name, **tool_args)
        
        return result
    
    def parse_tool_call(self, text):
        """解析 LLM 输出的工具调用"""
        # 简化解析，实际应该用结构化输出
        import re
        match = re.search(r'(\w+)\((.*?)\)', text)
        if match:
            return match.group(1), {'query': match.group(2)}
        return None, {}
```

**代码示例 - 向量记忆**：

```python
import numpy as np
from typing import List, Tuple

# Tom 的向量记忆系统
class VectorMemory:
    def __init__(self, embedding_model, dimension=768):
        self.embeddings = embedding_model
        self.memory = []  # (向量，内容) 对
        self.dimension = dimension
    
    def add(self, content: str):
        """添加记忆"""
        # 将内容转换为向量
        vector = self.embeddings.encode(content)
        self.memory.append((vector, content))
    
    def search(self, query: str, top_k=5) -> List[str]:
        """搜索相关记忆"""
        query_vector = self.embeddings.encode(query)
        
        # 计算相似度
        scores = []
        for vec, content in self.memory:
            score = self.cosine_similarity(query_vector, vec)
            scores.append((score, content))
        
        # 排序取 Top-K
        scores.sort(reverse=True)
        return [content for score, content in scores[:top_k]]
    
    def cosine_similarity(self, v1, v2):
        """计算余弦相似度"""
        return np.dot(v1, v2) / (np.linalg.norm(v1) * np.linalg.norm(v2))
```

```python
# Tom 的多智能体协作框架
class MultiAgentSystem:
    def __init__(self, agents):
        self.agents = agents  # {name: agent}
        self.message_board = []  # 公共消息板
    
    def broadcast(self, sender, message):
        """广播消息"""
        self.message_board.append({
            'sender': sender,
            'content': message
        })
        print(f"[{sender}]: {message}")
    
    def run(self, goal):
        """运行多智能体系统完成目标"""
        # 1. CEO 分配任务
        ceo_response = self.agents['CEO'].plan(goal)
        self.broadcast('CEO', ceo_response)
        
        # 2. 项目经理分解任务
        plan = self.agents['PM'].decompose(ceo_response)
        self.broadcast('项目经理', f"任务分解：{plan}")
        
        # 3. 执行智能体工作
        results = []
        for task in plan:
            # 根据任务类型选择执行智能体
            agent_type = self.classify_task(task)
            result = self.agents[agent_type].execute(task)
            results.append(result)
            self.broadcast(agent_type, f"完成：{task}")
        
        # 4. 质检
        quality_check = self.agents['QA'].review(results)
        self.broadcast('质检', quality_check)
        
        return results
    
    def classify_task(self, task):
        """根据任务类型分配智能体"""
        if '写' in task or '文档' in task:
            return 'Writer'
        elif '代码' in task or '程序' in task:
            return 'Coder'
        elif '图' in task or '设计' in task:
            return 'Designer'
        else:
            return 'General'
```
