Tom 的影像分析工具箱：

```python
import torch
import torch.nn as nn
from torchvision import models

# Tom 的医学影像分类器
class MedicalImageClassifier(nn.Module):
    def __init__(self, num_classes=3, pretrained=True):
        """
        使用 ResNet 作为 backbone
        
        Tom 的选择：
        - ResNet: 经典可靠
        - EfficientNet: 精度高
        - ViT: Transformer 方法
        """
        super().__init__()
        
        # 加载预训练模型
        self.backbone = models.resnet50(pretrained=pretrained)
        
        # 修改最后的全连接层
        num_features = self.backbone.fc.in_features
        self.backbone.fc = nn.Sequential(
            nn.Dropout(0.5),
            nn.Linear(num_features, 256),
            nn.ReLU(),
            nn.Linear(256, num_classes)
        )
    
    def forward(self, x):
        return self.backbone(x)

# Tom 的分割模型 (U-Net 风格)
class UNetSegmentation(nn.Module):
    def __init__(self, num_classes=2):
        """
        U-Net: 医学影像分割的黄金标准
        
        Tom 比喻：
        编码器：把图像压缩成特征
        解码器：把特征还原成 mask
        跳跃连接：保留细节信息
        """
        super().__init__()
        
        # 编码器 (下采样)
        self.enc1 = self._block(1, 64)   # 输入通道 1(灰度图)
        self.enc2 = self._block(64, 128)
        self.enc3 = self._block(128, 256)
        self.enc4 = self._block(256, 512)
        
        self.pool = nn.MaxPool2d(2, 2)
        
        # 瓶颈层
        self.bottleneck = self._block(512, 1024)
        
        # 解码器 (上采样)
        self.upconv4 = nn.ConvTranspose2d(1024, 512, 2, stride=2)
        self.dec4 = self._block(1024, 512)
        
        self.upconv3 = nn.ConvTranspose2d(512, 256, 2, stride=2)
        self.dec3 = self._block(512, 256)
        
        self.upconv2 = nn.ConvTranspose2d(256, 128, 2, stride=2)
        self.dec2 = self._block(256, 128)
        
        self.upconv1 = nn.ConvTranspose2d(128, 64, 2, stride=2)
        self.dec1 = self._block(128, 64)
        
        self.final = nn.Conv2d(64, num_classes, 1)
    
    def _block(self, in_channels, out_channels):
        return nn.Sequential(
            nn.Conv2d(in_channels, out_channels, 3, padding=1),
            nn.ReLU(inplace=True),
            nn.Conv2d(out_channels, out_channels, 3, padding=1),
            nn.ReLU(inplace=True)
        )
    
    def forward(self, x):
        # 编码器
        e1 = self.enc1(x)
        e2 = self.enc2(self.pool(e1))
        e3 = self.enc3(self.pool(e2))
        e4 = self.enc4(self.pool(e3))
        
        # 瓶颈
        b = self.bottleneck(self.pool(e4))
        
        # 解码器 (带跳跃连接)
        d4 = self.dec4(torch.cat([self.upconv4(b), e4], dim=1))
        d3 = self.dec3(torch.cat([self.upconv3(d4), e3], dim=1))
        d2 = self.dec2(torch.cat([self.upconv2(d3), e2], dim=1))
        d1 = self.dec1(torch.cat([self.upconv1(d2), e1], dim=1))
        
        return self.final(d1)
```

```python
from transformers import AutoTokenizer, AutoModelForTokenClassification
from transformers import TrainingArguments, Trainer
import torch

# Tom 的医学命名实体识别器
class MedicalNER:
    def __init__(self, model_name="cambridgeltl/BioBERT_base_v1.1"):
        """
        使用 BioBERT - 生物医学领域的 BERT
        
        Tom 的选择理由：
        - 在医学文本上预训练
        - 理解医学术语
        - 比通用 BERT 更准确
        """
        self.tokenizer = AutoTokenizer.from_pretrained(model_name)
        self.model = AutoModelForTokenClassification.from_pretrained(
            model_name, 
            num_labels=9  # BIO 标签：B-疾病，I-疾病，B-药物，I-药物，...
        )
        
        self.label_map = {
            0: 'O',      # 其他
            1: 'B-DIS',  # 疾病开始
            2: 'I-DIS',  # 疾病内部
            3: 'B-SYM',  # 症状开始
            4: 'I-SYM',  # 症状内部
            5: 'B-DRUG', # 药物开始
            6: 'I-DRUG', # 药物内部
            7: 'B-TEST', # 检查开始
            8: 'I-TEST', # 检查内部
        }
    
    def predict(self, text):
        """预测文本中的医学实体"""
        # 分词
        inputs = self.tokenizer(text, return_tensors="pt", truncation=True)
        
        # 预测
        with torch.no_grad():
            outputs = self.model(inputs)
            predictions = torch.argmax(outputs.logits, dim=-1)
        
        # 解析结果
        entities = self._extract_entities(
            inputs['input_ids'][0], 
            predictions[0], 
            text
        )
        
        return entities
    
    def _extract_entities(self, tokens, labels, original_text):
        """解析 BIO 标签为实体"""
        entities = []
        current_entity = None
        
        for i, (token, label_id) in enumerate(zip(tokens, labels)):
            label = self.label_map[label_id.item()]
            
            if label.startswith('B-'):
                # 开始新实体
                if current_entity:
                    entities.append(current_entity)
                entity_type = label[2:]
                current_entity = {
                    'type': entity_type,
                    'text': self.tokenizer.decode([token]),
                    'start': i
                }
            elif label.startswith('I-') and current_entity:
                # 继续当前实体
                current_entity['text'] += self.tokenizer.decode([token])
            else:
                # 结束当前实体
                if current_entity:
                    current_entity['end'] = i
                    entities.append(current_entity)
                    current_entity = None
        
        if current_entity:
            current_entity['end'] = len(tokens)
            entities.append(current_entity)
        
        return entities
```

知识图谱推理：

```python
# Tom 的医学知识图谱
class MedicalKnowledgeGraph:
    def __init__(self):
        """
        医学知识图谱
        
        Tom 的知识点：
        疾病 -[症状]-> 头痛
        疾病 -[检查]-> CT 扫描
        疾病 -[治疗]-> 药物
        药物 -[副作用]-> 恶心
        """
        self.graph = {}
    
    def add_relation(self, entity1, relation, entity2):
        """添加关系"""
        if entity1 not in self.graph:
            self.graph[entity1] = {}
        if relation not in self.graph[entity1]:
            self.graph[entity1][relation] = []
        self.graph[entity1][relation].append(entity2)
    
    def query(self, entity, relation=None):
        """查询实体的关系"""
        if entity not in self.graph:
            return []
        
        if relation is None:
            return self.graph[entity]
        
        return self.graph[entity].get(relation, [])
    
    def diagnose(self, symptoms):
        """
        基于症状推断可能的疾病
        
        Tom 的推理方法：
        1. 找到与每个症状相关的疾病
        2. 计算每个疾病的匹配分数
        3. 返回最可能的疾病
        """
        disease_scores = {}
        
        for symptom in symptoms:
            # 反向查询：哪些疾病有这个症状
            for disease, relations in self.graph.items():
                if '症状' in relations:
                    if symptom in relations['症状']:
                        disease_scores[disease] = disease_scores.get(disease, 0) + 1
        
        # 按分数排序
        sorted_diseases = sorted(
            disease_scores.items(), 
            key=lambda x: x[1], 
            reverse=True
        )
        
        return sorted_diseases
```
